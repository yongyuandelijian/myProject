odpscmd="/home/admin/odps/bin/odpscmd"
path='JC_GSZJ/J1_JSSJ/SC_JC_GSZJ/sql/'
# 获取内容进行读取，执行
for file in `ls ${path}`
do
sql_command=`cat ${path}${file}`
echo "正在执行的sql是：${sql_command}"
${odpscmd} --project="SC_JC_GSZJ" -e "${sql_command}"  >templog/${file}.log
done
